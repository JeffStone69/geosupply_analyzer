# ANSI color codes for beautiful console output
class Colors:
    RESET = "\033[0m"
    GREEN = "\033[92m"
    YELLOW = "\033[93m"
    BLUE = "\033[94m"
    CYAN = "\033[96m"
    RED = "\033[91m"
    BOLD = "\033[1m"


def print_colored(text: str, color: str = "white") -> None:
    """Print text with ANSI colors."""
    color_code = {
        "green": Colors.GREEN,
        "yellow": Colors.YELLOW,
        "blue": Colors.BLUE,
        "cyan": Colors.CYAN,
        "red": Colors.RED,
        "white": Colors.RESET,
    }.get(color.lower(), Colors.RESET)
    print(f"{color_code}{text}{Colors.RESET}")


def run_git_command(cmd: List[str], cwd: Path) -> bool:
    """Run a git command and return success status."""
    try:
        subprocess.run(["git"] + cmd, cwd=cwd, check=True, capture_output=True, text=True)
        return True
    except subprocess.CalledProcessError as e:
        print_colored(f"Git command failed: {' '.join(cmd)}", "red")
        print_colored(e.stderr.strip(), "red")
        return False
    except FileNotFoundError:
        print_colored("Git is not installed or not in PATH. Please install git.", "red")
        return False


def merge_gitignore(path: Path, standard_content: str) -> bool:
    """Safely merge standard .gitignore entries without destroying user changes."""
    if not path.exists():
        path.write_text(standard_content, encoding="utf-8")
        return True

    existing_lines = path.read_text(encoding="utf-8").splitlines(keepends=True)
    existing_set = {line.strip() for line in existing_lines if line.strip()}

    standard_lines = standard_content.splitlines(keepends=True)
    to_add = [
        line for line in standard_lines
        if line.strip() and line.strip() not in existing_set
    ]

    if to_add:
        with path.open("a", encoding="utf-8") as f:
            if existing_lines and not existing_lines[-1].endswith("\n"):
                f.write("\n")
            f.write("\n# === Best-practice ignores added by setup_repo.py ===\n")
            f.writelines(to_add)
        return True
    return False
