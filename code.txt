import streamlit as st
import yfinance as yf
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
from datetime import datetime, timedelta

# Set page configuration
st.set_page_config(page_title="GeoSupply Rebound Analyzer v9.2", page_icon="📈", layout="wide")

# Custom CSS for better styling
st.markdown("""
    <style>
    .main .block-container {
        padding-top: 2rem;
        padding-bottom: 2rem;
    }
    </style>
""", unsafe_allow_html=True)

# Title and description
st.title("GeoSupply Rebound Analyzer v9.2 (ASX Focused)")
st.markdown("Analyze supply chain rebound trends for ASX-listed companies with real-time data and interactive charts.")

# Cache data fetching to improve performance
@st.cache_data
def fetch_stock_data(ticker, start_date, end_date):
    """
    Fetch historical stock data for a given ticker using yfinance.
    Args:
        ticker (str): Stock ticker symbol (e.g., 'BHP.AX')
        start_date (str): Start date for data in 'YYYY-MM-DD' format
        end_date (str): End date for data in 'YYYY-MM-DD' format
    Returns:
        pd.DataFrame: Historical stock data
    """
    try:
        stock = yf.Ticker(ticker)
        df = stock.history(start=start_date, end=end_date)
        if df.empty:
            return None
        return df
    except Exception as e:
        st.error(f"Error fetching data for {ticker}: {str(e)}")
        return None

# Sidebar for user inputs
with st.sidebar:
    st.header("Filters")
    ticker = st.text_input("Enter ASX Ticker (e.g., BHP.AX)", value="BHP.AX")
    days_back = st.slider("Days of Historical Data", min_value=30, max_value=365, value=180)
    end_date = datetime.today()
    start_date = end_date - timedelta(days=days_back)
    st.markdown("---")
    st.caption("GeoSupply Rebound Analyzer v9.2")

# Fetch data based on user input
data = fetch_stock_data(ticker, start_date.strftime('%Y-%m-%d'), end_date.strftime('%Y-%m-%d'))

# Main content with tabs for different views
if data is not None:
    tab1, tab2 = st.tabs(["📊 Price Analysis", "📈 Rebound Metrics"])

    with tab1:
        st.subheader(f"Price History for {ticker}")
        # Plot closing price using Plotly
        fig_price = px.line(data, x=data.index, y='Close', title=f"Closing Price of {ticker}",
                            labels={'Close': 'Price (AUD)', 'Date': 'Date'})
        fig_price.update_layout(hovermode='x unified')
        st.plotly_chart(fig_price, use_container_width=True)

        # Display basic statistics
        st.subheader("Basic Statistics")
        st.write(data['Close'].describe())

    with tab2:
        st.subheader(f"Supply Chain Rebound Metrics for {ticker}")
        # Calculate simple moving averages for rebound analysis
        data['SMA_20'] = data['Close'].rolling(window=20).mean()
        data['SMA_50'] = data['Close'].rolling(window=50).mean()

        # Plot moving averages
        fig_ma = go.Figure()
        fig_ma.add_trace(go.Scatter(x=data.index, y=data['Close'], mode='lines', name='Close Price'))
        fig_ma.add_trace(go.Scatter(x=data.index, y=data['SMA_20'], mode='lines', name='20-Day SMA', line=dict(color='orange')))
        fig_ma.add_trace(go.Scatter(x=data.index, y=data['SMA_50'], mode='lines', name='50-Day SMA', line=dict(color='red')))
        fig_ma.update_layout(title="Price with Moving Averages (Rebound Indicator)",
                             xaxis_title="Date", yaxis_title="Price (AUD)", hovermode='x unified')
        st.plotly_chart(fig_ma, use_container_width=True)

        # Rebound signal (basic logic: when 20-day SMA crosses above 50-day SMA)
        data['Rebound_Signal'] = data['SMA_20'] > data['SMA_50']
        rebound_days = data[data['Rebound_Signal'] == True].index
        if not rebound_days.empty:
            st.markdown("**Rebound Signal Detected!**")
            st.write(f"Potential rebound signals on: {rebound_days[-1].strftime('%Y-%m-%d')} (latest)")
        else:
            st.markdown("**No Rebound Signal Detected.**")
else:
    st.warning("No data available for the selected ticker. Please check the ticker symbol or try again later.")

# Footer
st.markdown("---")
st.caption("Developed with Streamlit | Data sourced from Yahoo Finance | Not financial advice")
